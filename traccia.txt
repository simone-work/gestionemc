        const [result] = await promisePool.query(DELETE FROM ?? WHERE ?, [table, condition]);

    } catch (error) {
        console.error('Errore durante la DELETE:', error);
        res.status(500).send('Errore interno del server');
    }
});


const authenticateToken = (req, res, next) => {
    const token = req.cookies.jwtToken;
    if (!token) {
        return res.sendStatus(401); // Nessun token fornito
    }

    jwt.verify(token, process.env.ACCESS_TOKEN_SECRET, (err, user) => {
        if (err) {
            return res.sendStatus(403); // Token non valido o scaduto
        }
        req.user = user;
        next();
    });
};

TEORIA

- Il middleware prepara e pulisce i dati. 

- Il controller lavora con i dati già puliti. Il database salva la versione standardizzata e pulita dei dati.

- Differenza dalle vecchie applicazioni web, e che adesso il caricamento della pagina e costruzione viene eseguito dal browser del client e il server gli manda solo un file.js (prima un index.html, scarno con solo elemento script), e da
  quel file veranno generate le varie pagine senza avere un effettivo cambiamento della pagina.

          Aspetto	                  Vecchio Sistema (Server-Rendered)	                            Nuovo Sistema (Client-Rendered / SPA)
    Stato	                Senza stato (Stateless). Gestito con Sessioni/Cookie.	        Con stato (Stateful). Gestito nel client (Redux, etc.).
    Routing	                Gestito dal Server. Ogni URL è una nuova richiesta HTTP.	    Gestito dal Client. Nessun ricaricamento di pagina.
    Tooling	                Semplice (editor di testo, FTP, ricarica browser).	            Complesso e potente (npm, Vite, Hot-Reloading).
    SEO	                    Ottimo e nativo.                                                Problematico, richiede soluzioni come SSR (Next.js).
    Offline	Impossibile.    Richiede sempre una connessione.	                            Possibile. Base per le Progressive Web Apps (PWA).
